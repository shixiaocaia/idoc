    int n = nums.length;
    // æ„é€ å‰ç¼€å’Œ
    int[] sum = new int[n + 1];
    sum[0] = 0; 
    for (int i = 0; i < n; i++)
        sum[i + 1] = sum[i] + nums[i];
    
    int ans = 0;

        for (int j = 0; j < i; j++)

                ans++;

}
